=== VERSIÓN CON DEPURACIÓN AVANZADA ===
local Params = {
    RepoURL = "https://raw.githubusercontent.com/OkumaruSempaiSan/UwUsito/update-loadstring-format/",
    SSI = "saveinstance",
}

print("🔍 Iniciando carga de SaveInstance...")

-- Verificar si game:HttpGet existe
if not game or not game.HttpGet then
    print("❌ ERROR: game:HttpGet no está disponible")
    print("¿Estás ejecutando esto en Roblox?")
    return
end

-- Intentar descargar
local success, scriptContent = pcall(function()
    return game:HttpGet(Params.RepoURL .. Params.SSI .. ".luau", true)
end)

if not success then
    print("❌ ERROR al descargar:", scriptContent)
    
    -- Intentar con versión HTTP simple
    success, scriptContent = pcall(function()
        return game:HttpGet(Params.RepoURL .. Params.SSI .. ".lua", true)
    end)
    
    if not success then
        print("❌ Ambas descargas fallaron")
        return
    else
        print("✅ Versión .lua descargada (" .. #scriptContent .. " bytes)")
    end
else
    print("✅ Script .luau descargado (" .. #scriptContent .. " bytes)")
end

-- Intentar cargar el script
local loadSuccess, loadedFunction = pcall(function()
    return loadstring(scriptContent, Params.SSI)
end)

if not loadSuccess then
    print("❌ ERROR al parsear el script:", loadedFunction)
    return
end

print("✅ Script parseado correctamente")

-- Intentar ejecutar el script
local execSuccess, synsaveinstance = pcall(function()
    return loadedFunction()
end)

if not execSuccess then
    print("❌ ERROR al ejecutar el script:", synsaveinstance)
    return
end

print("🚀 SaveInstance cargado y listo!")

-- Configurar opciones
local Options = {
    Decompile = false,
    SaveBytecode = false,
    scriptcache = false,
    SafeMode = false,
    ShowStatus = false,
}

-- Ejecutar SaveInstance
local runSuccess, errorMsg = pcall(function()
    synsaveinstance(Options)
end)

if not runSuccess then
    print("❌ ERROR al ejecutar SaveInstance:", errorMsg)
else
    print("🎉 SaveInstance ejecutado exitosamente!")
    print("💡 Debería aparecer la interfaz de SaveInstance")
end
