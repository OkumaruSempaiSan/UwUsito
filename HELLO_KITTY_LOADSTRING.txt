-- 💖 HELLO KITTY PINK SAVEINSTANCE LOADER 💖
-- 🎀 Versión kawaii compacta (esquina derecha, se cierra automáticamente)

local Params = {
    RepoURL = "https://raw.githubusercontent.com/OkumaruSempaiSan/UwUsito/update-loadstring-format/",
    SSI = "saveinstance",
}

-- 🌸 Crear interfaz de carga kawaii COMPACTA
local function createCompactKawaiiUI()
    if not game:GetService("CoreGui") then return nil end
    
    local screenGui = Instance.new("ScreenGui")
    screenGui.Name = "KawaiiCompactLoader"
    screenGui.ZIndexBehavior = Enum.ZIndexBehavior.Sibling
    screenGui.Parent = game:GetService("CoreGui")
    
    -- 🐱 Contenedor principal COMPACTO (esquina superior derecha)
    local mainFrame = Instance.new("Frame")
    mainFrame.Name = "MainFrame"
    mainFrame.BackgroundColor3 = Color3.fromRGB(255, 105, 180) -- Rosa hot pink
    mainFrame.Position = UDim2.new(0.75, 0, 0.02, 0) -- Esquina derecha, cerca del borde superior
    mainFrame.Size = UDim2.new(0.23, 0, 0.08, 0) -- Mucho más pequeño
    mainFrame.Parent = screenGui
    
    -- ✨ Borde brillante
    local uiCorner = Instance.new("UICorner")
    uiCorner.CornerRadius = UDim.new(0.15, 0)
    uiCorner.Parent = mainFrame
    
    local uiStroke = Instance.new("UIStroke")
    uiStroke.Color = Color3.fromRGB(255, 255, 255) -- Blanco
    uiStroke.Thickness = 2
    uiStroke.Parent = mainFrame
    
    -- 🌸 Icono pequeño (corazón animado)
    local loadingIcon = Instance.new("TextLabel")
    loadingIcon.Name = "LoadingIcon"
    loadingIcon.Text = "💖"
    loadingIcon.TextColor3 = Color3.fromRGB(255, 255, 255)
    loadingIcon.TextScaled = true
    loadingIcon.Font = Enum.Font.Fantasy
    loadingIcon.BackgroundTransparency = 1
    loadingIcon.Position = UDim2.new(0.05, 0, 0.1, 0)
    loadingIcon.Size = UDim2.new(0.15, 0, 0.8, 0)
    loadingIcon.Parent = mainFrame
    
    -- 🎀 Mensaje de estado COMPACTO
    local statusLabel = Instance.new("TextLabel")
    statusLabel.Name = "StatusLabel"
    statusLabel.Text = "🌸 Cargando..."
    statusLabel.TextColor3 = Color3.fromRGB(255, 255, 255)
    statusLabel.TextScaled = true
    statusLabel.Font = Enum.Font.Cartoon
    statusLabel.BackgroundTransparency = 1
    statusLabel.Position = UDim2.new(0.25, 0, 0.1, 0)
    statusLabel.Size = UDim2.new(0.7, 0, 0.8, 0)
    statusLabel.Parent = mainFrame
    
    -- 📊 Barra de progreso mini
    local progressBarBack = Instance.new("Frame")
    progressBarBack.Name = "ProgressBarBack"
    progressBarBack.BackgroundColor3 = Color3.fromRGB(255, 255, 255) -- Blanco
    progressBarBack.Position = UDim2.new(0.02, 0, 0.85, 0)
    progressBarBack.Size = UDim2.new(0.96, 0, 0.1, 0)
    progressBarBack.Parent = mainFrame
    
    local progressBarCorner = Instance.new("UICorner")
    progressBarCorner.CornerRadius = UDim.new(0.5, 0)
    progressBarCorner.Parent = progressBarBack
    
    local progressBar = Instance.new("Frame")
    progressBar.Name = "ProgressBar"
    progressBar.BackgroundColor3 = Color3.fromRGB(255, 20, 147) -- Rosa profundo
    progressBar.Position = UDim2.new(0, 0, 0, 0)
    progressBar.Size = UDim2.new(0, 0, 1, 0)
    progressBar.Parent = progressBarBack
    
    local progressBarCorner2 = Instance.new("UICorner")
    progressBarCorner2.CornerRadius = UDim.new(0.5, 0)
    progressBarCorner2.Parent = progressBar
    
    -- Animación de corazón pulsante
    local heartSize = 1
    local heartDirection = 1
    game:GetService("RunService").Heartbeat:Connect(function(delta)
        heartSize = heartSize + (delta * 2 * heartDirection)
        if heartSize > 1.3 then
            heartDirection = -1
        elseif heartSize < 0.9 then
            heartDirection = 1
        end
        loadingIcon.TextSize = 18 * heartSize
    end)
    
    -- Función para actualizar estado
    local function updateStatus(text, progress)
        statusLabel.Text = text
        if progress then
            progressBar.Size = UDim2.new(progress, 0, 1, 0)
        end
    end
    
    -- Función para cerrar CON animación
    local function closeUI()
        -- Animación de desvanecimiento
        for i = 1, 0, -0.1 do
            mainFrame.BackgroundTransparency = 1 - i
            uiStroke.Transparency = 1 - i
            statusLabel.TextTransparency = 1 - i
            loadingIcon.TextTransparency = 1 - i
            progressBarBack.BackgroundTransparency = 1 - i
            progressBar.BackgroundTransparency = 1 - i
            task.wait(0.02)
        end
        screenGui:Destroy()
    end
    
    return {
        UpdateStatus = updateStatus,
        Close = closeUI,
        GUI = screenGui
    }
end

-- 🎀 Función principal kawaii MEJORADA
local function loadKawaiiSaveInstance()
    print("🌸 Iniciando Hello Kitty SaveInstance...")
    
    -- Crear interfaz de carga compacta
    local loadingUI = createCompactKawaiiUI()
    if not loadingUI then
        print("❌ No se pudo crear la interfaz (CoreGui no disponible)")
        -- Continuar sin interfaz
        loadWithoutUI()
        return
    end
    
    -- Paso 1: Descargar script
    loadingUI.UpdateStatus("📥 Descargando...", 0.2)
    local scriptContent
    local success, err = pcall(function()
        scriptContent = game:HttpGet(Params.RepoURL .. Params.SSI .. ".luau", true)
        return scriptContent
    end)
    
    if not success then
        loadingUI.UpdateStatus("❌ Error descarga", 0)
        task.wait(1)
        loadingUI.Close()
        return
    end
    
    loadingUI.UpdateStatus("✅ Descargado", 0.4)
    task.wait(0.3)
    
    -- Paso 2: Cargar script
    loadingUI.UpdateStatus("🎀 Cargando...", 0.6)
    local loadSuccess, loadedFunc = pcall(function()
        return loadstring(scriptContent, Params.SSI)
    end)
    
    if not loadSuccess then
        loadingUI.UpdateStatus("❌ Error carga", 0)
        task.wait(1)
        loadingUI.Close()
        return
    end
    
    loadingUI.UpdateStatus("💖 Parseando...", 0.8)
    task.wait(0.3)
    
    -- Paso 3: Ejecutar script
    local execSuccess, synsaveinstance = pcall(function()
        return loadedFunc()
    end)
    
    if not execSuccess then
        loadingUI.UpdateStatus("❌ Error ejecución", 0)
        task.wait(1)
        loadingUI.Close()
        return
    end
    
    loadingUI.UpdateStatus("🎉 Listo! 🐱", 1.0)
    task.wait(0.5)
    
    -- Cerrar interfaz ANTES de ejecutar SaveInstance
    loadingUI.Close()
    
    -- Configurar opciones
    local Options = {
        Decompile = false,
        SaveBytecode = false,
        scriptcache = false,
        SafeMode = false,
        ShowStatus = false,
    }
    
    print("💖 Ejecutando SaveInstance...")
    
    -- Ejecutar SaveInstance (esto abrirá SU propia interfaz)
    local runSuccess, runError = pcall(function()
        synsaveinstance(Options)
    end)
    
    if not runSuccess then
        print("❌ Error al ejecutar SaveInstance:", runError)
    else
        print("🎉 SaveInstance ejecutado exitosamente!")
    end
end

-- 🎀 Versión alternativa sin interfaz (si CoreGui no está disponible)
local function loadWithoutUI()
    print("🌸 Cargando SaveInstance sin interfaz...")
    
    -- Descargar script
    local scriptContent
    local success, err = pcall(function()
        return game:HttpGet(Params.RepoURL .. Params.SSI .. ".luau", true)
    end)
    
    if not success then
        print("❌ Error al descargar:", err)
        return
    end
    
    print("✅ Script descargado (" .. #scriptContent .. " bytes)")
    
    -- Cargar script
    local loadSuccess, loadedFunc = pcall(function()
        return loadstring(scriptContent, Params.SSI)
    end)
    
    if not loadSuccess then
        print("❌ Error al cargar:", loadedFunc)
        return
    end
    
    -- Ejecutar script
    local execSuccess, synsaveinstance = pcall(function()
        return loadedFunc()
    end)
    
    if not execSuccess then
        print("❌ Error al ejecutar:", synsaveinstance)
        return
    end
    
    -- Configurar opciones
    local Options = {
        Decompile = false,
        SaveBytecode = false,
        scriptcache = false,
        SafeMode = false,
        ShowStatus = false,
    }
    
    -- Ejecutar SaveInstance
    local runSuccess, runError = pcall(function()
        synsaveinstance(Options)
    end)
    
    if not runSuccess then
        print("❌ Error final:", runError)
    else
        print("🎉 SaveInstance cargado!")
    end
end

-- 🚀 Iniciar automáticamente (con timeout de seguridad)
task.spawn(function()
    loadKawaiiSaveInstance()
end)

-- ⏰ Timeout de seguridad (si se queda atascado)
task.wait(10)
print("⏰ Timeout: Si no ves la interfaz de SaveInstance, algo salió mal.")
